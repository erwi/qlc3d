# it mirrors the source directory's "tests" directory and same relative paths can be used for accessing
# files in tests.

file(COPY ${CMAKE_CURRENT_SOURCE_DIR}/python DESTINATION ${CMAKE_CURRENT_BINARY_DIR}) # copy python scripts
